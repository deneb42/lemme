#Le compilateur c++ utilisé par le Makefile. Par défaut le compilateur est g++. Si vous en changez, merci de vous assurer que votre code compile aussi avec g++.
CPP = g++

#Les options du compilateur c++
#Pensez à mettre -g et enlever -O3 pour le debug!
CPPOPT = -O3 -Wall

#Les options de l'éditeur de lien
LNKOPT = 

all: occurences
test: ./occurences

occurences: Lemme.o main.o
	$(CPP) $(LNKOPT) Lemme.o main.o -o occurences

Lemme.o: Lemme.cpp
	$(CPP) $(CPPOPT) -c Lemme.cpp -o Lemme.o

main.o: main.cpp
	$(CPP) $(CPPOPT) -c main.cpp -o main.o

mrproper: clean
	rm occurences

clean:
	rm Lemme.o main.o
